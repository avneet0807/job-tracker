⚡ Connect Google Sheets — Permanent Storage + Daily Reminders

Step 1

Create your Google Sheet

Go to sheets.google.com → Create a new blank sheet → Name it "Job Tracker". Leave it empty — the script will set it up automatically.

Step 2

Add the Apps Script backend

In your Google Sheet, go to Extensions → Apps Script. Delete any existing code and paste this entire script:

// Avi's Job Tracker — Google Apps Script Backend // Paste this entire script into Extensions > Apps Script in your Google Sheet const SHEET_NAME = 'Jobs'; const EMAIL = Session.getActiveUser().getEmail(); function doPost(e) { try { const data = JSON.parse(e.postData.contents); const action = data.action; if (action === 'saveAll') return respond(saveAll(data.jobs)); if (action === 'getAll') return respond(getAll()); return respond({error: 'Unknown action'}); } catch(err) { return respond({error: err.message}); } } function doGet(e) { try { return respond(getAll()); } catch(err) { return respond({error: err.message}); } } function respond(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); const headers = ['id','company','role','salary','location','reporting','status', 'contact','recruiterCompany','email','phone','url','appliedDate','followup','notes','added','lastUpdated']; sheet.appendRow(headers); sheet.getRange(1,1,1,headers.length).setFontWeight('bold'); sheet.setFrozenRows(1); } return sheet; } function getAll() { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return {jobs: []}; const headers = data[0]; const jobs = data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i] || ''); return obj; }); return {jobs}; } function saveAll(jobs) { const sheet = getSheet(); const headers = ['id','company','role','salary','location','reporting','status', 'contact','recruiterCompany','email','phone','url','appliedDate','followup','notes','added','lastUpdated']; // Clear existing data (keep header) const lastRow = sheet.getLastRow(); if (lastRow > 1) sheet.deleteRows(2, lastRow - 1); // Write all jobs if (jobs && jobs.length > 0) { const rows = jobs.map(j => headers.map(h => j[h] || '')); sheet.getRange(2, 1, rows.length, headers.length).setValues(rows); } return {success: true, count: jobs ? jobs.length : 0}; } // Daily email digest — set this to run every morning function sendDailyDigest() { const result = getAll(); const jobs = result.jobs || []; const today = new Date(); today.setHours(0,0,0,0); const followupsDue = []; const overdue = []; jobs.forEach(j => { if (!j.followup || j.status === 'Rejected' || j.status === 'Offer') return; const fDate = new Date(j.followup); fDate.setHours(0,0,0,0); const diffDays = Math.round((fDate - today) / (1000*60*60*24)); if (diffDays < 0) overdue.push({...j, diffDays}); else if (diffDays <= 7) followupsDue.push({...j, diffDays}); }); // Also flag applied > 7 days with no update const stalled = jobs.filter(j => { if (j.status !== 'Applied') return false; if (!j.appliedDate) return false; const applied = new Date(j.appliedDate); const diffDays = Math.round((today - applied) / (1000*60*60*24)); return diffDays >= 7; }); if (followupsDue.length === 0 && overdue.length === 0 && stalled.length === 0) return; let html = `

📋 Job Tracker — Daily Digest

`; if (overdue.length > 0) { html += `

⚠️ Overdue Follow-ups (${overdue.length})

`; overdue.forEach((j,i) => { html += ``; }); html += `
RoleCompanyContactDays Overdue
${j.role}${j.company}${j.contact||'—'}${Math.abs(j.diffDays)} days
`; } if (followupsDue.length > 0) { html += `

🗓 Follow-ups Due This Week (${followupsDue.length})

`; followupsDue.forEach((j,i) => { html += ``; }); html += `
RoleCompanyContactDue In
${j.role}${j.company}${j.contact||'—'} ${j.phone?'('+j.phone+')':''} ${j.diffDays===0?'Today':j.diffDays+' day'+( j.diffDays!==1?'s':'')}
`; } if (stalled.length > 0) { html += `

⏸ Stalled Applications (Applied 7+ days ago)

`; stalled.forEach((j,i) => { html += ``; }); html += `
RoleCompanyAppliedRecruiter
${j.role}${j.company}${j.appliedDate}${j.contact||'—'}
`; } html += `

Sent by your Job Tracker · ${new Date().toLocaleDateString('en-AU')}

`; MailApp.sendEmail({ to: EMAIL, subject: `📋 Job Tracker Digest — ${followupsDue.length + overdue.length} follow-up${followupsDue.length+overdue.length!==1?'s':''} due`, htmlBody: html }); } // Run once to set up the daily trigger function setupDailyTrigger() { // Delete existing triggers ScriptApp.getProjectTriggers().forEach(t => ScriptApp.deleteTrigger(t)); // Create new daily trigger at 8am ScriptApp.newTrigger('sendDailyDigest') .timeBased() .everyDays(1) .atHour(8) .create(); return 'Daily digest trigger set for 8am!'; }
Step 3

Deploy as Web App

In Apps Script: click Deploy → New Deployment → Type: Web App → Execute as: Me → Who has access: Anyone → Click Deploy → Copy the Web App URL.

Step 4

Set up daily email reminders

In Apps Script, select the function setupDailyTrigger from the dropdown and click Run. This sets up an 8am daily email digest automatically.
